2 # this feature now there is no need to require <tt>ActiveRecord::Base</tt> at boot time purely to apply configuration. Instead

20 module ActiveSupport
 
21   @load_hooks = Hash.new {|h,k| h[k] = [] }
 
22   @loaded = {}

24   def self.on_load(name, options = {}, &block)
 
25     if base = @loaded[name]
