277       #   Strings:: Some content to evaluate
 
278       #   Procs::   A proc to call with the object
 
279       #   Objects:: An object with a before_foo method on it to call

292       def _compile_filter(filter)
 
293         method_name = "_callback_#{@kind}_#{next_id}"

451       # Set callbacks for a previously defined callback.
 
452       #
 
453       # Syntax:
 
454       #   set_callback :save, :before, :before_meth
 
455       #   set_callback :save, :after,  :after_meth, :if => :condition
 
456       #   set_callback :save, :around, lambda { |r| stuff; yield; stuff }
 
457       #
 
458       # Use skip_callback to skip any defined one.
 
459       #
 
460       # When creating or skipping callbacks, you can specify conditions that
 
461       # are always the same for a given key. For instance, in Action Pack,
 
462       # we convert :only and :except conditions into per-key conditions.
 
463       #
 
464       #   before_filter :authenticate, :except => "index"
 
465       #
 
466       # becomes
 
467       #
 
468       #   dispatch_callback :before, :authenticate, :per_key => {:unless => proc {|c| c.action_name == "index"}}
 
469       #
 
470       # Per-Key conditions are evaluated only once per use of a given key.
 
471       # In the case of the above example, you would do:
 
472       #
 
473       #   run_callbacks(:dispatch, action_name) { ... dispatch stuff ... }
 
474       #
 
475       # In that case, each action_name would get its own compiled callback
 
476       # method that took into consideration the per_key conditions. This
 
477       # is a speed improvement for ActionPack.
 
478       #
 
479       def set_callback(name, *filter_list, &block)
 
480         mapped = nil
 
481 
482         __update_callbacks(name, filter_list, block) do |chain, type, filters, options|
 
483           mapped ||= filters.map do |filter|
 
484             Callback.new(chain, filter, type, options.dup, self)
 
485           end
 
486 
487           filters.each do |filter|
 
488             chain.delete_if {|c| c.matches?(type, filter) }
 
489           end
 
490 
491           options[:prepend] ? chain.unshift(*(mapped.reverse)) : chain.push(*mapped)
 
492         end
 
493       end

498         __update_callbacks(name, filter_list, block) do |chain, type, filters, options|

528       # Defines callbacks types:
 
529       #
 
530       #   define_callbacks :validate
 
531       #
 
532       # This macro accepts the following options:
 
533       #
 
534       # * <tt>:terminator</tt> - Indicates when a before filter is considered

536       #
 
537       #   define_callbacks :validate, :terminator => "result == false"
 
538       #

540       # other callbacks are not executed. Defaults to "false", meaning no value
 
541       # halts the chain.
 
542       #
 
543       # * <tt>:rescuable</tt> - By default, after filters are not executed if
 
544       # the given block or a before filter raises an error. Set this option to
 
545       # true to change this behavior.
 
546       #
 
547       # * <tt>:scope</tt> - Indicates which methods should be executed when a class
 
548       # is given as callback. Defaults to <tt>[:kind]</tt>.

551       #    def before(caller)

573       # In the above case whenever you save an account the method <tt>Audit#before</tt> will
 
574       # be called. On the other hand
