79     # Returns a number of degrees from due north (clockwise).

135     # Compute the geographic center (aka geographic midpoint, center of
 
136     # gravity) for an array of geocoded objects and/or [lat,lon] arrays

138     # the procedure documented at http://www.geomidpoint.com/calculation.html.

140     def geographic_center(points)
 
141

143       coords = points.map{ |p| to_radians(extract_coordinates(p)) }
 
144 
145       # convert to Cartesian coordinates
 
146       x = []; y = []; z = []
 
147       coords.each do |p|
 
148         x << Math.cos(p[0]) * Math.cos(p[1])
 
149         y << Math.cos(p[0]) * Math.sin(p[1])
 
150         z << Math.sin(p[0])
 
151       end
